403 页面，配合自定义操作。

````jsx
import Exception from 'ant-design-pro/lib/Exception';
import { Button } from 'antd';

const actions = (
  <div>
    <Button type="primary" size="large">回到首页</Button>
    <Button size="large">查看详情</Button>
  </div>
);
ReactDOM.render(
  <Exception type="403" actions={actions} />
, mountNode);
````
